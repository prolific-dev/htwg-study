002package einstieg;
003
004import java.util.Scanner;
005
006/**
007*Einstieg ist ein kleines Java-Programm.
008*Es verwendet die Klassen Scanner und System aus der Java-Bibliothek.
009*Dennis Agostinho da Silva
010*16.10.2018
011*/
012public final class Einstieg {
013    private Einstieg() { }
014
015    private static final Scanner EINGABE = new Scanner(System.in);
016
017/**
018*Die Klassenmethode main ist der Startpunkt des Programms.
019*main verwendet die Methoden print und printf zum Ausgeben von Text
020*sowie die Methoden next und nextInt zum Einlesen von Text
021*und einer ganzen Zahl.
022*@param args wird nicht verwendet
023*/
024    public static void main(final String[] args) {
025        System.out.print("Vorname: ");
026        String vorname = EINGABE.next();
027
028        System.out.print("Anzahl bisher geschriebener Java-Programme: ");
029        int anzahl = EINGABE.nextInt();
030
031        System.out.printf("%ss %d. Java-Programm funktioniert!%n",
032            vorname, anzahl + 1);
033
034    }
035}
