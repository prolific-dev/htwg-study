003package aufgabe1;
004
005import java.util.Scanner;
006
007/**
008 * IntVar zeigt den Umgang mit Variablen vom Typ int.
009 * &Uuml;bungsaufgabe 1 zur Programmiertechnik 1.
010 * Dennis Agostinho da Silva
011 * 23.10.2018
012 */
013public final class IntVar {
014    private IntVar() { }
015
016    private static final Scanner EINGABE = new Scanner(System.in);
017
018    /**
019     * main ist der Startpunkt des Programms.
020     * @param args wird nicht verwendet.
021     */
022    public static void main(final String[] args) {
023
024        final int max = Integer.MAX_VALUE;
025        final int min = Integer.MIN_VALUE;
026
027        System.out.printf("Zwei ganze Zahlen zwischen %d und %d eingeben: %n",
028             max, min);
029
030
031        int zahl1 = EINGABE.nextInt();
032        int zahl2 = EINGABE.nextInt();
033
034
035        System.out.printf("%d%n ist oktal %o%n und hexadezimal %x%n",
036            zahl1, zahl1, zahl1);
037        System.out.printf("%d%n ist oktal %o%n und hexadezimal %x%n",
038            zahl2, zahl2, zahl2);
039
040
041
042        int summe = zahl1 + zahl2;
043        int minus = zahl1 - zahl2;
044        int multi = zahl1 * zahl2;
045        int divis = zahl1 / zahl2;
046        int modul = zahl1 % zahl2;
047        boolean bool1 = false;
048        boolean bool2 = true;
049
050        System.out.println(zahl1 + " + " + zahl2 + " ist " + summe);
051        System.out.println(zahl1 + " - " + zahl2 + " ist " + minus);
052        System.out.println(zahl1 + " * " + zahl2 + " ist " + multi);
053        System.out.println(zahl1 + " / " + zahl2 + " ist " + divis);
054        System.out.println(zahl1 + " % " + zahl2 + " ist " + modul);
055
056
057
058        if (zahl1 == zahl2) {
059            System.out.println(zahl1 + " == " + zahl2 + " ist " + bool2);
060        } else {
061            System.out.println(zahl1 + " == " + zahl2 + " ist " + bool1);
062        }
063
064
065        if (zahl1 != zahl2) {
066            System.out.println(zahl1 + " != " + zahl2 + " ist " + bool2);
067        } else {
068            System.out.println(zahl1 + " != " + zahl2 + " ist " + bool1);
069        }
070
071
072        if (zahl1 < zahl2) {
073            System.out.println(zahl1 + " < " + zahl2 + " ist " + bool2);
074        } else {
075            System.out.println(zahl1 + " < " + zahl2 + " ist " + bool1);
076        }
077
078
079        if (zahl1 <= zahl2) {
080            System.out.println(zahl1 + " <= " + zahl2 + " ist " + bool2);
081        } else {
082            System.out.println(zahl1 + " <= " + zahl2 + " ist " + bool1);
083        }
084
085
086        if (zahl1 > zahl2) {
087            System.out.println(zahl1 + " > " + zahl2 + " ist " + bool2);
088        } else {
089            System.out.println(zahl1 + " > " + zahl2 + " ist " + bool1);
090        }
091
092
093        if (zahl1 >= zahl2) {
094            System.out.println(zahl1 + " >= " + zahl2 + " ist " + bool2);
095        } else {
096            System.out.println(zahl1 + " >= " + zahl2 + " ist " + bool1);
097        }
098
099
100    }
101}
