002package aufgabe2;
003import java.util.Random;
004
005/**
006 * Wuerfel gibt Zufallszahlen zwischen 1 und 6 aus.
007 * @author H.Drachenfels
008 * @version 15.5.2018
009 */
010public final class Wuerfel {
011    private Wuerfel() { }
012
013    private static final int ANZAHL_AUGEN = 6;
014
015    /**
016     * main ist der Startpunkt des Programms.
017     * @param args Anzahl der Zufallszahlen (Default 1)
018     */
019    public static void main(final String[] args) {
020        int versuche = 1;
021
022        if (args.length > 0) {
023            try {
024                versuche = Integer.parseInt(args[0]);
025            } catch (NumberFormatException x) {
026                versuche = 0;
027            }
028
029            if (versuche <= 0 || args.length > 1) {
030                System.err.printf("Zu viele oder falsche Parameter!%n"
031                                  + "Aufruf: java Wuerfel [Anzahl]%n");
032                return;
033            }
034        }
035
036        Random wuerfel = new Random();
037        for (int i = 0; i < versuche; ++i) {
038            System.out.println(wuerfel.nextInt(ANZAHL_AUGEN) + 1);
039        }
040    }
041}
